def validate_birth_date(year, month, day, hour, minute):
    """Validate birth date"""
    try:
        # Check year range
        if year < 1900 or year > 2100:
            return False, "Year must be between 1900-2100"
        
        # Check month
        if month < 1 or month > 12:
            return False, "Month must be between 1-12"
        
        # Check day
        if day < 1 or day > 31:
            return False, "Day must be between 1-31"
        
        # Check hour
        if hour < 0 or hour > 23:
            return False, "Hour must be between 0-23"
        
        # Check minute
        if minute < 0 or minute > 59:
            return False, "Minute must be between 0-59"
        
        # Check specific date validity
        try:
            datetime.datetime(year, month, day, hour, minute)
        except ValueError:
            return False, "Invalid date time"
        
        return True, "Validation passed"
        
    except Exception as e:
        return False, f"Validation error: {str(e)}"

@app.route('/api/calculate', methods=['POST'])
def calculate():
    """Bazi calculation API"""
    try:
        data = request.get_json()
        
        if not data:
            return jsonify({
                "success": False,
                "error": "Invalid request data",
                "message": "Please provide birth date time"
            }), 400
        
        # Get parameters
        year = data.get('year')
        month = data.get('month')
        day = data.get('day')
        hour = data.get('hour', 0)
        minute = data.get('minute', 0)
        longitude = data.get('longitude', 120.0)
        latitude = data.get('latitude', 30.0)
        
        # Validate parameters
        if not all([year, month, day]):
            return jsonify({
                "success": False,
                "error": "Missing required parameters",
                "message": "Please provide year, month, day"
            }), 400
        
        # Validate date
        is_valid, message = validate_birth_date(year, month, day, hour, minute)
        if not is_valid:
            return jsonify({
                "success": False,
                "error": "Date validation failed",
                "message": message
            }), 400
        
        # Create datetime object
        birth_time = datetime.datetime(year, month, day, hour, minute)
        
        # Calculate Bazi
        result = calculate_bazi(birth_time, longitude, latitude)
        
        # Add request info
        result["request_info"] = {
            "birth_date": f"{year}-{month:02d}-{day:02d}",
            "birth_time": f"{hour:02d}:{minute:02d}",
            "location": f"Longitude {longitude}, Latitude {latitude}"
        }
        
        return jsonify(result)
        
    except Exception as e:
        return jsonify({
            "success": False,
            "error": str(e),
            "message": "Server internal error"
        }), 500

            const bazi = data.bazi;
            document.getElementById('yearPillar').textContent = bazi['Year Pillar'];
            document.getElementById('monthPillar').textContent = bazi['Month Pillar'];
            document.getElementById('dayPillar').textContent = bazi['Day Pillar'];
            document.getElementById('hourPillar').textContent = bazi['Hour Pillar'];
            document.getElementById('fullBazi').textContent = 
                `${bazi['Year Pillar']} ${bazi['Month Pillar']} ${bazi['Day Pillar']} ${bazi['Hour Pillar']}`;
            
            // Display Five Elements
            const elements = data.elements;
            const elementsGrid = document.getElementById('elementsGrid');
            elementsGrid.innerHTML = '';
            
            for (const [position, element] of Object.entries(elements)) {
                const item = document.createElement('div');
                item.className = 'element-item';
                item.innerHTML = `
                    <div style="font-weight: bold; margin-bottom: 5px;">${position}</div>
                    <div style="font-size: 20px; color: #667eea;">${element}</div>
                `;
                elementsGrid.appendChild(item);
            }
            
            // Display Ten Gods analysis
            const tenGods = data.ten_gods_analysis;
            const tableBody = document.querySelector('#tenGodsTable tbody');
            tableBody.innerHTML = '';
            
            for (const [position, info] of Object.entries(tenGods)) {
                const row = document.createElement('tr');
                row.innerHTML = `
                    <td>${position}</td>
                    <td>${info.Stem}</td>
                    <td>${info.Branch}</td>
                    <td>${info.Element}</td>
                    <td>${info['Ten God']}</td>
                `;
                tableBody.appendChild(row);
